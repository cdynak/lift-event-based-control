\subsection{States}

\[
s_i = [d_i, o_i, f_i, m_i]
\]
where:

\(d \in \{0,1\}\)
\begin{itemize}
  \item \(0\) - down
  \item \(1\) - up
\end{itemize}

\(o \in \{0,1\}\)
\begin{itemize}
  \item \(0\) - closed
  \item \(1\) - open
\end{itemize}

\(f \in \{0,1,...,i,...,n\}\)
\begin{itemize}
  \item \(i\) - last reached floor
  \item \(n\) - number of floors
\end{itemize}

\(S=[s_1, s_2, ..., s_l]\)
\begin{itemize}
  \item \(l\) - number of lifts
\end{itemize}

\[
W = [S, P, Q]

\subsection{Events}

\[
E = E^l \cup E^c \cup E^x \cup E^i
\]

\[
E^i = [\text{lift\_nr},\text{button\_nr}]
\]

\(\text{direction} \in \{0,1\}\)

\[
E^x = [\text{lift\_nr}, \text{command}]
\]

\(\text{command} \in \{0,1,2,3,4\}\)

\[
E^c = [\text{lift\_nr}, \text{ACK}]
\]

\(\text{ACK} \in \{0,1,2,3\}\)
